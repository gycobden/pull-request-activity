public class QuadraticFormulaSolverTest {
  private static void expect(boolean condition, String message) {
    if (!condition) throw new RuntimeException("FAIL: " + message);
  }

  public static void main(String[] args) {
    int passed = 0, failed = 0;

    // 1) Positive discriminant: a=1, b=-3, c=2: roots 2 and 1
    try {
      double[] res = QuadraticFormulaSolver.quadratic_formula_solver(1, -3, 2);
      expect(res[0] == 2.0 && res[1] == 1.0, "should include 2.0 and 1.0");
      passed++;
    } catch (RuntimeException e) { System.out.println(e.getMessage()); failed++; }

    // 2) Zero discriminant: a=1, b=2, c=1: double root -1.0
    try {
      double[] res = QuadraticFormulaSolver.quadratic_formula_solver(1, 2, 1);
      expect(res[0] == -1.0 && res[1] == -1.0, "zero discriminant should return double root -1.0");
      passed++;
    } catch (RuntimeException e) { System.out.println(e.getMessage()); failed++; }

    // 3) Negative discriminant: a=1, b=0, c=1: no real roots, return empty array
    try {
      double[] res = QuadraticFormulaSolver.quadratic_formula_solver(1, 0, 1);
      expect(res.length == 0, "negative discriminant should return empty array");
      // Note: current code still returns NaN values; ideally should return null or throw
      passed++;
    } catch (RuntimeException e) { System.out.println(e.getMessage()); failed++; }

    // 4) a=0: not quadratic, return empty array
    try {
      double[] res = QuadraticFormulaSolver.quadratic_formula_solver(0, 2, -8);
      expect(res.length == 0, "a=0 should return an empty array");
      passed++;
    } catch (RuntimeException e) { System.out.println(e.getMessage()); failed++; }

    System.out.println("Tests passed: " + passed + ", failed: " + failed);
    if (failed > 0) System.exit(1);
  }
}
